import numpy as np

def region_of_interest(img, vertices):

def drow_the_lines(img, lines):
    img = np.copy(img)

def process(image):                                      #This function overwrites a frame with the the detected lines
    print(image.shape)
    height = image.shape[0]

        (0, height),
        (width/2, height/2),
        (width, height)
    ]

                            lines=np.array([]),
